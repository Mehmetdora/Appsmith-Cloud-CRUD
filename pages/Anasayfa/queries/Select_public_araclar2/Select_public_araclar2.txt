SELECT
  id,marka,model,yil,public.kategoriler.kat_isim,kategori_id
FROM
  public.araclar
INNER JOIN public.kategoriler ON public.kategoriler.kat_id = public.araclar.kategori_id
WHERE
  "marka" ilike '%{{araba_table.searchText}}%'
ORDER BY
  "{{araba_table.sortOrder.column || 'id'}}" {{araba_table.sortOrder.order !== "desc" ? "" : "DESC"}}
LIMIT
  {{araba_table.pageSize}}
OFFSET
  {{araba_table.pageOffset}}
